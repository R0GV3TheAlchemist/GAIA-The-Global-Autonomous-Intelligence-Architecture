activator of this non-visual cognitive channel.

  geniculate nucleus, amygdala, and prefrontal cortex** — regions governing decision-making,

import numpy as np

wavelengths = np.linspace(380, 700, 1000)  # nm

def gaussian(x, mu, sigma, amp=1.0):
    return amp * np.exp(-0.5 * ((x - mu) / sigma) ** 2)

# Photoreceptor spectral sensitivity curves (modeled on published opsin data)
melanopsin = gaussian(wavelengths, 480, 40, 1.0)   # ipRGC non-image-forming cognition
s_cone     = gaussian(wavelengths, 430, 35, 0.8)   # alertness, mood
m_cone     = gaussian(wavelengths, 535, 45, 0.85)  # spatial clarity
l_cone     = gaussian(wavelengths, 565, 50, 0.75)  # arousal, grounding

# Filter definitions (Gaussian bandpass on white-light carrier)
filters = {
    "white":  np.ones_like(wavelengths),
    "violet": gaussian(wavelengths, 415, 30),
    "blue":   gaussian(wavelengths, 470, 35),
    "green":  gaussian(wavelengths, 525, 40),
    "yellow": gaussian(wavelengths, 575, 35),
    "red":    gaussian(wavelengths, 650, 35),
}

def spectral_score(filt, sensitivity):
    return float(np.trapz(filt * sensitivity, wavelengths))

for name, filt in filters.items():
    alertness  = 0.5 * spectral_score(filt, melanopsin) + \
                 0.3 * spectral_score(filt, s_cone) + \
                 0.2 * spectral_score(filt, m_cone)

                 0.3 * spectral_score(filt, s_cone) + \
                 0.3 * spectral_score(filt, melanopsin)
    grounding  = 0.6 * spectral_score(filt, l_cone) + \
                 0.4 * spectral_score(filt, m_cone)
    coherence  = spectral_score(filt, np.ones_like(wavelengths))
    print(f"{name}: alertness={alertness:.1f}, clarity={clarity:.1f}, "
          f"grounding={grounding:.1f}, coherence={coherence:.1f}")
